\begin{alignat}{2}

        && \qquad \text{Atomic Packing Factor} \\[10pt]
    & E \approx \frac{dF_N}{dr}
        && \qquad \text{Elasticity (Force vs Interatomic distance)} \\[10pt]
    & \sigma = \frac{F}{A_0}
        && \qquad \text{Engineering Stress} \\[10pt]
    & \epsilon = \frac{\Delta L}{L_0}
        && \qquad \text{Engineering Strain} \\[10pt]
    & \sigma_T = \frac{F}{A}
        && \qquad \text{True Stress (Instantaneous Area)} \\[10pt]
    & \epsilon_T = \ln\left(\frac{L}{L_0}\right)
        && \qquad \text{True Strain} \\[10pt]
    & G = H - TS 
        && \qquad \text{Gibbs Free Energy (Enthalpy Form)} \\[10pt]
    & G = \sum_{i=1}^{j} n_i G_i 
        && \qquad \text{Total Gibbs Energy (Multi-phase)} \\[10pt]
    & W_\alpha + W_L = 1 
        && \qquad \text{Total Mass Balance} \\[10pt]
    & W_\alpha C_\alpha + W_L C_L = C_o 
        && \qquad \text{Component Mass Balance} \\[10pt]
    & W_\alpha = \frac{C_L - C_o}{C_L - C_\alpha} 
        && \qquad \text{Solid Fraction } (\alpha) \\[10pt]
    & W_L = \frac{C_o - C_\alpha}{C_L - C_\alpha} 
        && \qquad \text{Liquid Fraction } (L)
\end{alignat}
